6   package tetris.logiikka;
7   
8   import java.awt.Color;
9   import tetris.domain.GameSituation;
10  import tetris.gui.TetrisFrame;
11  
12  /**
13   * Implementations of Communicator are capable of communication between gui and
14   * GameSituation.
15   *
16   * @author isjani
17   */
18  public interface Communicator {
19      
20      /**
21       * starts a new game.
22       */
23      public void newGame();
24  
25      /**
26       * Sets the engine to communicate with. (for creation of a rematch
27       * purposes).
28       *
29       * @param engine engine to communicate with.
30       */
31      public void setEngine(Engine engine);
32  
33      /**
34       * Reads quotes from a source and tells the frame to use them for a
35       * distraction board.
36       */
37      public void setDistractionBoard();
38  
39      /**
40       * Retrieves highest score from text file, and gives it to the frame.
41       */
42      public void setHighScore();
43  
44  
45      public int getScore();
46  
47      /**
48       * Attempts to move the piece to given direction.
49       *
50       * @param direction to move
51       */
52      public void movePiece(Direction direction);
53  
54      /**
55       * Attempts to rotate the piece according to given rotation.
56       *
57       * @param rotation direction of rotation
58       */
59      public void rotatePiece(Rotation rotation);
60  
61      /**
62       * Checks whether the game is active or not.
63       * @return true if game is active
64       */
65      public boolean gameIsActive();
66  
67      /**
68       * Sets the frame to which we want to communicate.
69       *
70       * @param frame to communicate with
71       */
72      public void setFrame(TetrisFrame frame);
73  
74      /**
75       * Sets the GameSituation with which the communication is to be done.
76       *
77       * @param gameSituation to communicate with
78       */
79      public void setGameSituation(GameSituation gameSituation);
80  
81      /**
82       * Sets the PieceDropper with which the communication is to be done.
83       *
84       * @param pieceDropper to communicate with
85       */
86      public void setPieceDropper(PieceDropper pieceDropper);
87  
88  /**
89   * Tells the frame to repaint itself.
90   * @param colorTable to use for repainting.
91   */
92      public void rePaintSituation(Color[][] colorTable);
93  
94      /**
95       * Tells the PieceDropper to stop when it is running, or continue if it's
96       * stopped.
97       */
98      public void pauseGame();
99  }
