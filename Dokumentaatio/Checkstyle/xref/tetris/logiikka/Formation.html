12  public enum Formation {
13      O(new int[]{0, 1, 1, 0}, new int[]{0, 0, 1, 1}, -1), //(0,0) is top left of the square
14      L(new int[]{0, 1, 0, 0}, new int[]{0, 0, -1, -2}, 2), //(0,0) is bottom left of standing L
15      T(new int[]{0, 0, 1, -1}, new int[]{0, 1, 1, 1}, 1), //(0,0) is the top piece of T lying down on its long back
16      S(new int[]{0, -1, 1, 0}, new int[]{0, 0, -1, -1}, 3), //(0,0) is the bottom right of S
17      I(new int[]{0, -1, 1, 2}, new int[]{0, 0, 0, 0}, 0), // (0,0)  is the second left piece of I lying down
18      Z(new int[]{0, 0, -1, 1}, new int[]{0, -1, -1, 0}, 1), // (0,0) is the bottom left block
19      J(new int[]{0, -1, 0, 0}, new int[]{0, 0, -1, -2}, 2); // (0,0) is  is bottom right of standing J// (0,0) is  is bottom right of standing J
20  
21      public int[] xVal;
22      public int[] yVal;
23      public int pivotIndex;
24  
25      private Formation(int[] x, int[] y, int p) {
26          this.xVal = x;
27          this.yVal = y;
28          this.pivotIndex = p;
29      }
30  }
