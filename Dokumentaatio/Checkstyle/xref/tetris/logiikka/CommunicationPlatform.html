6   package tetris.logiikka;
7   
8   import java.awt.Color;
9   import tetris.domain.Block;
10  import tetris.domain.MoveResult;
11  import tetris.domain.GameSituation;
12  import tetris.gui.TetrisFrame;
13  
14  /**
15   * This class receives messages from threads and acts upon GameSituation (based
16   * on messages). After every communication, asks to draw the gui again.
17   *
18   * @author isjani
19   */
20  public class CommunicationPlatform implements Communicator {
21  
22      private GameSituation gs;
23      private TetrisFrame frame;
24  
25      public CommunicationPlatform() {
26      }
27  
28      /**
29       * {@inheritDoc}
30       */
31      public synchronized void movePiece(Direction direction) {
32          MoveResult moveResult = gs.movePiece(direction);
33          actBasedOnResult(moveResult);
34      }
35  
36      /**
37       * {@inheritDoc}
38       */
39      public void setGameSituation(GameSituation gs) {
40          this.gs = gs;
41      }
42  
43      /**
44       * {@inheritDoc}
45       */
46      public synchronized void rotatePiece(Rotation rotation) {
47          gs.rotatePiece(rotation);
48          Color[][] colorTable = blockTableToColorTable(gs.getFieldAndPieceBlocks());
49          rePaintSituation(colorTable);
50      }
51  
52      private Color[][] blockTableToColorTable(Block[][] blocks) {
53          int rowCount = blocks.length;
54          int columnCount = blocks[1].length;
55          Color[][] colorTable = new Color[rowCount][columnCount];
56          for (int row = 0; row < rowCount; row++) {
57              for (int column = 0; column < columnCount; column++) {
58                  Block block = blocks[row][column];
59                  if (block == null) {
60                      colorTable[row][column] = Color.BLACK;
61                  } else {
62                      colorTable[row][column] = block.getColor();
63                  }
64              }
65          }
66          return colorTable;
67      }
68  
69      private void actBasedOnResult(MoveResult moveResult) {
70          if (moveResult.pieceWasMoved) {
71              rePaintSituation(blockTableToColorTable(gs.getFieldAndPieceBlocks()));
72          } else if (moveResult.gameWon) {
73              victory();
74          } else if (moveResult.gameLost) {
75              defeat();
76          } else if (moveResult.pieceWasFrozen) {
77              gs.createActivePiece();
78              rePaintSituation(blockTableToColorTable(gs.getFieldAndPieceBlocks()));
79          }
80      }
81  
82      private void defeat() {
83          gs.gameIsActive = false;
84      }
85  
86      private void victory() {
87          gs.gameIsActive = false;
88      }
89  
90      /**
91       * {@inheritDoc}
92       */
93      public synchronized boolean gameIsActive() {
94          return gs.gameIsActive;
95      }
96  
97      /**
98       * {@inheritDoc}
99       */
100     public void setFrame(TetrisFrame frame) {
101         this.frame = frame;
102     }
103 
104     /**
105      * {@inheritDoc}
106      */
107     public void rePaintSituation(Color[][] colorTable) {
108         frame.rePaintSituation(colorTable);
109     }
110 
111 }
