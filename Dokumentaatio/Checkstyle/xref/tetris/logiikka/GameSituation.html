16  public class GameSituation {

18      private int width;
19      private int height;
20  
21      private Field field;
22      private Piece activePiece;
23  
24      public GameSituation(int width, int height) {
25          this.width = width;
26          this.height = height;
27          field = new Field(width, height);
28      }
29  
30      /*
31      Moves the piece to the given direction if the required spots are vacant in the field.
32      If there is no vacancy and the direction is down, the piece will be frozen to the field.
33      If there is no vacancy and the direction is not down, the piece will not move.
34      
35      Returns true,true if the piece was moved and game continues, or piece wasn't allowed to move and game continues.
36      Returns false, true if piece was frozen and game continues  --- requires new piece
37      Returns false false if piece wasn't moved and game doesn't continue --- requires game to stop
38       */
39      public boolean[] movePiece(Direction dir) {
40          /* Checking if the piece can be moved to the direction */
41          List<Block> blocks = activePiece.getBlocks();
42          for (int i = 0; i < 4; i++) {
43              Block b = blocks.get(i);
44              if (!field.spotIsVacant(b.getX() + dir.x, b.getY() + dir.y)) {
45                  if (dir == Direction.DOWN) {
46                      if (!field.freezePiece(activePiece)) {
47                          return new boolean[]{false, false};
48                      } else {
49                          return new boolean[]{false, true};
50                      }
51                  }
52                  return new boolean[]{true, true};
53              }
54          }
55          /* Checking done, the piece can move */
56          activePiece.move(dir);
57          return new boolean[]{true, true};
58      }
59

61          this.activePiece = p;
62      }
