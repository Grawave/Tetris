6   package tetris.domain;
7   
8   import java.awt.Color;
9   import java.util.ArrayList;
10  import java.util.Collections;
11  import java.util.List;
12  import java.util.Random;
13  
14  /**
15   * Describes how the blocks of given formation should be in relationship to each
16   * other.
17   *
18   * @author isjani
19   */
20  public enum Formation {
21      /**
22       * Square. No pivot. color?
23       */
24      O(new int[]{0, 1, 1, 0}, new int[]{0, 0, 1, 1}, -1, Color.RED), //(0,0) is top left of the square
25      /**
26       * L-shape. Pivot is the second highest block of a standing L.
27       */
28      L(new int[]{0, 1, 0, 0}, new int[]{0, 0, -1, -2}, 2, Color.BLUE), //(0,0) is bottom left of standing L
29      /**
30       * T-shape. Pivot is top middle block of a standing T.
31       */
32      T(new int[]{0, 0, 1, -1}, new int[]{0, 1, 1, 1}, 1, Color.CYAN), //(0,0) is the top piece of T lying down on its long back
33      /**
34       * S-shape. Pivot is the top middle block of a S.
35       */
36      S(new int[]{0, -1, 1, 0}, new int[]{0, 0, -1, -1}, 3, Color.ORANGE), //(0,0) is the bottom right of S
37      /**
38       * I-shape. Pivot is the second block from the left of I lying down (in the
39       * moment of creation).
40       */
41      I(new int[]{0, -1, 1, 2}, new int[]{0, 0, 0, 0}, 0, Color.WHITE), // (0,0)  is the second left piece of I lying down
42      /**
43       * Z-shape. Pivot is the top middle block of a Z
44       */
45      Z(new int[]{0, 0, -1, 1}, new int[]{0, -1, -1, 0}, 1, Color.YELLOW), // (0,0) is the bottom left block
46      /**
47       * J-shape. Pivot is the second highest block of a standing J.
48       */
49      J(new int[]{0, -1, 0, 0}, new int[]{0, 0, -1, -2}, 2, Color.MAGENTA); // (0,0) is  is bottom right of standing J// (0,0) is  is bottom right of standing J
50  
51      /**
52       * x values of block coordinates. x[i] corresponds to y[i]
53       */
54      public int[] xVal;
55      /**
56       * y values of block coordinates. y[i] corresponds to x[i]
57       */
58      public int[] yVal;
59      /**
60       * Starting coordinates of the pivot can be accessed by x[pivotIndex],
61       * y[pivotIndex].
62       */
63      public int pivotIndex;
64      /**
65       * The color unique to the blocks of this formation.
66       */
67      public Color color;
68      private static List<Formation> forms;
69  
70      private Formation(int[] x, int[] y, int p, Color color) {
71          this.xVal = x;
72          this.yVal = y;
73          this.pivotIndex = p;
74          this.color = color;
75      }
76  
77      /**
78       * Generates and returns a random object from the pool of all possible
79       * formations.
80       *
81       * @return A random Formation
82       */
83      public static Formation getRandom() {
84          if (forms == null) {
85              forms = new ArrayList<>();
86              Collections.addAll(forms, Formation.values());
87          }
88          Random random = new Random();
89          int randomIndex = random.nextInt(forms.size());
90          return forms.get(randomIndex);
91      }
92  }
