6   package tetris.gui;
7   
8   import java.awt.Color;
9   import java.awt.Component;
10  import java.awt.Dimension;
11  import java.awt.GridLayout;
12  import javax.swing.BorderFactory;
13  import javax.swing.JButton;
14  import javax.swing.JLabel;
15  import javax.swing.JPanel;
16  
17  /**
18   * This panel is for the actual game visuals. It contains a grid of JLabels that
19   * are colored according to input, creating the pieces and the field for the
20   * player.
21   *
22   * @author isjani
23   */
24  public class GameSituationPanel extends JPanel {
25  
26      private Component[][] components;
27      private int width;
28      private int height;
29      private final Color BACKGROUND_COLOR = Color.BLACK;
30      private final Color BORDER_COLOR = Color.GRAY;
31      private GridLayout layout;
32  
33      /**
34       * Creates a game situation panel and prepares it to have components
35       * according to the given width and height.
36       *
37       * @param width width of the field.
38       * @param height height of the field.
39       */
40      public GameSituationPanel(int width, int height) {
41          components = new JLabel[height][width];
42          this.width = width;
43          this.height = height;
44      }
45  
46      /**
47       * Repaints the field displayed to the player according to the given
48       * colorTable.
49       *
50       * @param colorTable components will be painted with this input.
51       */
52      public void rePaintSituation(Color[][] colorTable) {
53          for (int row = 0; row < height; row++) {
54              for (int column = 0; column < width; column++) {
55                  Component label = components[row][column];
56                  label.setBackground(colorTable[row][column]);
57              }
58          }
59      }
60  
61      private void createLayout() {
62          layout = new GridLayout(height, width);
63          setLayout(layout);
64      }
65  
66      /**
67       * Initializes the grid with the background color.
68       */
69      public void initialize() {
70          createLayout();
71          for (int row = 0; row < height; row++) {
72              for (int column = 0; column < width; column++) {
73                  JLabel label = new JLabel();
74                  label.setBackground(BACKGROUND_COLOR);
75                  label.setOpaque(true);
76                  label.setBorder(BorderFactory.createLineBorder(BORDER_COLOR));
77                  components[row][column] = label;
78                  add(label);
79              }
80          }
81      }
82  }
