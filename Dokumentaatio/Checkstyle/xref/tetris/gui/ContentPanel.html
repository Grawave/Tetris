35      private final Dimension GS_MIN = new Dimension(300, 600);

37      private final Dimension GS_MAX = new Dimension(500, 1100);
38      private Communicator communicator;
39      private JLabel scoreLabel;

41      private JLayeredPane leftLayeredPane;

49      /**
50       * Initializes the panel with a ScoreBoard and sets up keybindings.
51       *
52       * @param communicator to communicate with.
53       */
54      public ContentPanel(Communicator communicator) {
55          super();
56          this.communicator = communicator;
57          createLayout();
58          createScoreBoard();
59          createKeyBindings();
60          rightLayeredPane = new JLayeredPane();
61          add(rightLayeredPane);
62      }

64      /**
65       * Creates a panel that contains a background image and distractive quotes.
66       * Because why not?
67       *
68       * @param quotes quotes to be displayed.
69       */
70      public void createDistractionBoard(String[] quotes) {
71          rightLayeredPane.setLayout(null);
72          rightLayeredPane.setPreferredSize(GS_PREF);
73  
74          this.quotes = quotes;
75          quoteSettings();
76  
77          ImageIcon image = new ImageIcon(PIC_URL);
78          JLabel imageHolder = new JLabel();
79          imageHolder.setIcon(image);
80          imageHolder.setBounds(0, 0, 800, 1200);
81  
82          rightLayeredPane.add(imageHolder, 0, 0);
83          rightLayeredPane.add(quoteArea, 4, 4);
84  
85      }

87      private String newRandomQuote() {
88          Random random = new Random();
89          int index = random.nextInt(quotes.length - 1);
90          String quote = quotes[index];
91          return modifyToLines(quote);
92      }
93  
94      private String modifyToLines(String string) {
95          String modified = "Distractions for your mind.. \n  \n";
96          boolean lineChange = false;
97          int i = 0;
98          while (i < string.length()) {
99              char ch = string.charAt(i);
100             if (i % 30 == 0 && i != 0) {
101                 lineChange = true;
102             }
103             if (lineChange && ch == ' ') {
104                 lineChange = false;
105                 modified += "\n";
106             }
107             modified += ch;
108             i++;
109         }
110         return modified;
111     }

113     private void updateQuote() {
114         rightLayeredPane.remove(quoteArea);
115         quoteSettings();
116 
117         rightLayeredPane.add(quoteArea, 1, 1);
118         rightLayeredPane.repaint();
119     }
120 
121     private void quoteSettings() {
122         quoteArea = new JTextArea(newRandomQuote());
123         quoteArea.setBounds(0, 0, 600, 1000);
124         quoteArea.setFont(new Font("Serif", Font.PLAIN, 18));
125         quoteArea.setOpaque(false);
126         quoteArea.setForeground(Color.white);
127     }
128 
129     private void createScoreBoard() {
130         leftLayeredPane = new JLayeredPane();
131         leftLayeredPane.setLayout(null);
132         leftLayeredPane.setPreferredSize(GS_PREF);
133 
134         ImageIcon image = new ImageIcon(PIC_URL);
135         JLabel imageHolder = new JLabel();
136         imageHolder.setIcon(image);
137         imageHolder.setBounds(0, 0, 800, 1200);
138 
139         leftLayeredPane.add(imageHolder, 0, 0);
140 
141         scoreLabel = new JLabel("<html>SCORE<br>" + communicator.getScore() + "</html>");
142         scoreLabel.setForeground(Color.WHITE);
143         scoreLabel.setFont(new Font("Serif", Font.PLAIN, 30));
144         scoreLabel.setBounds(100, 10, 300, 300);
145 
146         leftLayeredPane.add(scoreLabel, 1, 1);
147 
148         add(leftLayeredPane);
149     }
150 
151     /**
152      * Updates the score status and puts a new random quote on the distraction
153      * board.
154      */
155     public void updateBoards() {
156         int score = communicator.getScore();
157         if (score == 0) {
158             return;
159         }
160         leftLayeredPane.remove(scoreLabel);
161         leftLayeredPane.repaint();
162 
163         scoreLabel = new JLabel("<html>SCORE<br>" + score + "</html>");
164         scoreLabel.setForeground(Color.WHITE);
165         scoreLabel.setFont(new Font("Serif", Font.PLAIN, 30));
166         scoreLabel.setBounds(100, 10, 300, 300);
167 
168         leftLayeredPane.add(scoreLabel, 1, 1);
169         leftLayeredPane.repaint();
170         updateQuote();
171     }

173     private void createKeyBindings() {
174         InputMap iPM = this.getInputMap();
175         ActionMap aCM = this.getActionMap();
176 
177         iPM.put(KeyStroke.getKeyStroke("LEFT"), "LEFT");
178         aCM.put("LEFT", new MoveAction(Direction.LEFT, communicator));
179 
180         iPM.put(KeyStroke.getKeyStroke("RIGHT"), "RIGHT");
181         aCM.put("RIGHT", new MoveAction(Direction.RIGHT, communicator));
182 
183         iPM.put(KeyStroke.getKeyStroke("DOWN"), "DOWN");
184         aCM.put("DOWN", new MoveAction(Direction.DOWN, communicator));
185 
186         iPM.put(KeyStroke.getKeyStroke("Z"), "ROTATE_LEFT");
187         aCM.put("ROTATE_LEFT", new RotateAction(Rotation.LEFT, communicator));
188 
189         iPM.put(KeyStroke.getKeyStroke("C"), "ROTATE_RIGHT");
190         aCM.put("ROTATE_RIGHT", new RotateAction(Rotation.RIGHT, communicator));
191 
192         iPM.put(KeyStroke.getKeyStroke("P"), "PAUSE");
193         aCM.put("PAUSE", new PauseAction(communicator));
194     }
195 
196     private void createLayout() {
197         layout = new GridLayout(0, 3);
198         setLayout(layout);
199     }

201     /**
202      * Adds the GameSituationPanel to this ContentPanel. Sets it's dimensions
203      *
204      * @param gs GameSituationPanel to be added.
205      */
206     public void addGS(GameSituationPanel gs) {
207         add(gs, 1);
208         gs.setPreferredSize(GS_PREF);
209         gs.setMinimumSize(GS_MIN);
210         gs.setMaximumSize(GS_MAX);
211     }

224     }
